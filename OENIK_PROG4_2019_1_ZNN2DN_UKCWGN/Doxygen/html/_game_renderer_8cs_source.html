110  public Drawing GetTile(Tile tile, int i, int j)
111  {
112  int x = (i * (this.width / this.Model.Gamesize)) + 5;
113  int y = (j * (this.width / this.Model.Gamesize)) + 100;
114 
115  int tileSize = this.width / this.Model.Gamesize;
116 
117  if (tile != null)
118  {
119  DrawingGroup dg = new DrawingGroup();

121  this.ValueToBackgroundBrush(tile.Value),
122  new Pen(Brushes.Black, 2),
123  new RectangleGeometry(new Rect(x, y, tileSize, tileSize), 15, 15)));
124 
125  FontFamilyConverter ffc = new FontFamilyConverter();
126  FontFamily fontFamily = (FontFamily)ffc.ConvertFromString("Imprint MT Shadow");
127  Typeface typeface = new Typeface(fontFamily, FontStyles.Normal, FontWeights.Bold, FontStretches.Condensed);
128 
129  int fontsize = 25;
130  if (this.Model.Gamesize == 3)
131  {
132  fontsize = 45;
133  }
134 
135  if (this.Model.Gamesize == 4)
136  {
137  fontsize = 35;
138  }
139 
140  Brush letterColor = this.ValueToForgroundBrush(tile.Value);
141 
142  FormattedText fm = new FormattedText(
143  tile.Value.ToString(),
144  System.Globalization.CultureInfo.CurrentCulture,
145  FlowDirection.LeftToRight,
146  typeface,
147  fontsize,
148  letterColor);
149 
150  dg.Children.Add(new GeometryDrawing(
151  letterColor,
152  new Pen(letterColor, 1),
153  fm.BuildGeometry(new Point(
154  x + (tileSize / 2) - (fm.Width / 2),
155  y + (tileSize / 2) - (fm.Height / 2)))));
156 
157  return dg;
158  }
159 
160  return new GeometryDrawing(
161  this.ValueToBackgroundBrush(0),
162  new Pen(Brushes.Black, 2),
163  new RectangleGeometry(new Rect(x, y, tileSize, tileSize), 15, 15));
164  }

54  public bool Move(int countdownFrom, int yIncr, int xIncr, int side)
55  {
56  bool moved = false;
57  for (int i = 0; i < side * side; i++)
58  {
59  int j = Math.Abs(countdownFrom - i);
60 
61  // we use this two variable for moving between tiles.
62  int r = j / side;
63  int c = j % side;
64 
65  // if the tile is empty we should continue.
66  if (this.GameModel.Board[r, c] == null)
67  {
68  continue; // passes control to the next iteration of the enclosing while, do, for, or foreach statement in which it appears.
69  }
70 
71  // we make a move to the specified deriction
72  int nextR = r + yIncr;
73  int nextC = c + xIncr;
74 
75  // while the tiles can move or cominate with each other.
76  while (nextR >= 0 && nextR < side && nextC >= 0 && nextC < side)
77  {
78  Tile next = this.GameModel.Board[nextR, nextC];
79  Tile curr = this.GameModel.Board[r, c];
80 
81  // if newyt tile is null, we dontd do any combination , just move there
82  if (next == null)
83  {
84  if (this.GameModel.CheckAvailableMoves)
85  {
86  return true;
87  }
88 
89  this.GameModel.Board[nextR, nextC] = curr;
90  this.GameModel.Board[r, c] = null;
91  r = nextR;
92  c = nextC;
93  nextR += yIncr;
94  nextC += xIncr;
95  moved = true;
96  }
97 
98  // if merge is possible with the next tile we should combinate them and move then
99  else if (next.CanMergeWith(curr))
100  {
101  if (this.GameModel.CheckAvailableMoves)
102  {
103  return true;
104  }
105 
106  // if the given tile has a bigger value than highest value we should overwrite the highest value.
107  int value = next.MergeWith(curr);
108  if (value > this.GameModel.Highest)
109  {
110  this.GameModel.Highest = value;
111  }
112 
113  // after a move we incrase the score.
114  this.GameModel.Score += value;
115  this.GameModel.Board[r, c] = null;
116  moved = true;
117  break;
118  }
119 
120  // else exit from the sequence.
121  else
122  {
123  break;
124  }
125  }
126  }
